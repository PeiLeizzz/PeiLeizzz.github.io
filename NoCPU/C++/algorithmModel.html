int binarySearch(int left, int right, int target) {
    while (left <= right) {
        int mid = left + right >> 1;
        if (a[mid] < target) left = mid + 1;
        else if (a[mid] > target) right = mid - 1;
        else return mid;
    }
    return -1;
}

# 冒泡排序 

void bubbleSort(int a[], int n) {
	for (int i = 0; i < n - 1; i++) {
		int flag = true;
		for (int j = 0; j < n - i - 1; j++) {
			if (a[j] > a[j+1]) {
				swap(a[j], a[j+1]);
				flag = false;
			}
		}
		if (flag) return;
	}
}

void insertSort(int a[], int n) {
	for (int i = 1; i < n; i++) {
		int x = a[i], j = i;
		while (j > 0 && a[j - 1] > x) 
			a[j] = a[--j];
		a[j] = x;
	}
}

# 选择排序 

void selectSort(int a[], int n) {
	for (int i = 0; i < n - 1; i++) {
		int m = i;
		for (int j = i + 1; j < n; j++) {
			if (a[j] < a[m]) {
				m = j;
			}
		}
		if (m != i) swap(a[i], a[m]);
	}
}

# 快速排序 

void quickSort(int a[], int l, int r) {
	if (l >= r) return;
	int i = l, j = r, pivot = a[i];
	while (i < j) {
		while (i < j && a[j] >= pivot) {
			j--;
		}
		if (i < j) a[i++] = a[j];
		while (i < j && a[i] <= pivot) {
			i++;
		} 
		if (i < j) a[j--] = a[i];
	}
	a[i] = pivot;
	quickSort(a, l, i-1);
	quickSort(a, i+1, r);
}

# 归并排序 

void mergeSort(int a[], int l, int r) {
	if (l >= r) return;
	int mid = l + r >> 1;
	mergeSort(a, l, mid);
	mergeSort(a, mid + 1, r);
	
	int tmp[r - l + 1], i = l, j = mid + 1, k = 0;
	while (i <= mid && j <= r) {
		while (i <= mid && a[i] <= a[j]) tmp[k++] = a[i++];
		while (j <= r && a[j] <= a[i]) tmp[k++] = a[j++];
	}
	while (i <= mid) tmp[k++] = a[i++];
	while (j <= r) tmp[k++] = a[j++];
	for (int i = 0; i < r - l + 1; i++) {
		a[i + l] = tmp[i];
	}
}

    LL quickPow(LL a, LL b, LL mod) {
    LL res = 0;
    while (b) {
        if (b & 1) {
            res = res * a % mod;
        }
        b >>= 1;
        a = a * a % mod;
    }

LL quickMul(LL a, LL b, LL mod) {
	LL res = 0;
	while (b) {
		if (b & 1) {
			res = (res + a) % mod;
		}
		b >>= 1;
		a = (a << 1) % mod; 
	}
	return res;
}

struct bign {
	int len;
	int d[1000];
	bign() {
		len = 0;
		memset(d, 0, sizeof(d));
	}
	bool operator<(bign a) {
		if (len < a.len) return true;
		else if (len > a.len) return false;
		else {
			for (int i = len - 1; i >= 0; i--) {
				if (d[i] > a.d[i]) return false;
				else if (d[i] < a.d[i]) return true;
			}
			return false;
		}
	}

// str: 1234 ==> bign 4321
bign convertToBign(char str[]) {
	bign res;
	res.len = strlen(str);
	for (int i = 0; i < res.len; i++) {
		res.d[i] = str[res.len - i - 1] - '0';
	}
	return res;
}

bign add(bign a, bign b) {
	bign res;
	int carry = 0, i = 0;
	int la = a.len, lb = b.len;
	while (i < la || i < lb) {
		int sum = a.d[i] + b.d[i] + carry;
		res.d[i++] = sum % 10;
		carry = sum / 10;
	}
	if (carry != 0) res.d[i++] = carry;
	res.len = i;
	return res;
}

bign sub(bign a, bign b) {
	bign res;
	int la = a.len, lb = b.len;
	int i = 0;
	while (i < la || i < lb) {
		if (a.d[i] >= b.d[i]) {
			res.d[i] = a.d[i] - b.d[i];
		}
		else {
			a.d[i+1]--;
			res.d[i] = a.d[i] + 10 - b.d[i];
		}
		i++;
	}
	// 去除高位 0 并保证至少有一位数字
	while (i > 1 && res.d[i-1] == 0) {
		i--;
	}
	res.len = i;
	return res;
}

bign mul(bign a, int b) {
	bign res;
	int carry = 0, i = 0;
	int la = a.len;
	while (i < la) {
		int sum = a.d[i] * b + carry;
		res.d[i++] = sum % 10;
		carry = sum / 10;
	}
	while (carry > 0) {
		res.d[i++] = carry % 10;
		carry /= 10;
	}
	res.len = i;
	return res;
}

bign div(bign a, int b) {
	bign res;
	int carry = 0, i = a.len - 1;
	int la = a.len;
	while (i >= 0) {
		carry = carry * 10 + a.d[i];
		if (carry < b) {
			res.d[i] = 0;
		}
		else {
			res.d[i] = carry / b;
			carry %= b;
		}
		i--;
	}
	i = la;
	while (i > 1 && res.d[i-1] == 0) {
		i--;
	}
	res.len = i;
	return res;
}

# 分数计算

class Fraction {
public:
	ll up, down;
	
	Fraction reduction(Fraction a) {
		if (a.up == 0) {
			a.down = 1;
			return a;
		}
		else if (a.down < 0) {
			a.up = -a.up;
			a.down = -a.down;
		}
		ll L = gcd(abs(a.up), abs(a.down));
		a.up /= L;
		a.down /= L;
		return a;
	}
	
	Fraction operator+(Fraction b) {
		Fraction res;
		res.up = this->up * b.down + this->down * b.up;
		res.down = this->down * b.down;
		res = res.reduction(res);
		return res;
	}
	
	Fraction operator-(Fraction b) {
		Fraction res;
		res.up = this->up * b.down - this->down * b.up;
		res.down = this->down * b.down;
		res = res.reduction(res);
		return res;
	}
	
	Fraction operator*(Fraction b) {
		Fraction res;
		res.up = this->up * b.up;
		res.down = this->down * b.down;
		res = res.reduction(res);
		return res;
	}
	
	Fraction operator/(Fraction b) {
		Fraction res;
		if (b.up == 0) {
			res.up = 0, res.down = 0;
			printf("除数不能为 0\n");
			return res;
		}
		res.up = this->up * b.down;
		res.down = this->down * b.up;
		res = res.reduction(res);
		return res;
	}
	
	void print(Fraction a) {
		a = a.reduction(a);
		if (a.down == 1) printf("%lld\n", a.up);
		else if (abs(a.up) > a.down) {
			printf("%lld %lld/%lld\n", a.up / a.down, abs(a.up) % a.down, a.down);
		}
		else {
			printf("%lld/%lld\n", a.up, a.down);
		}
	}
};

struct TreeNode {
	TreeNode* left;
	TreeNode* right;
	int key, value, n;
	TreeNode(int k, int v, int N) {
		key = k;
		value = v;
		n = N;
		left = NULL;
		right = NULL;
	}
};

class BST {
private:
	TreeNode* root;
	
	int size(TreeNode* node) {
		if (node == NULL) return 0;
		return node->n;
	}
	
	int getValue(TreeNode* node, int key) {
		if (node == NULL) return -1;
		if (key < node->value) return getValue(node->left, key);
		else if (key > node->value) return getValue(node->right, key);
		return node->value;
	}
	
	TreeNode* put(TreeNode* node, int key, int value) {
		if (node == NULL) return new TreeNode(key, value, 1);
		if (key < node->value) node->left = put(node->left, key, value);
		else if (key > node->value) node->right = put(node->right, key, value);
		else node->value = value;
		node->n = size(node->left) + size(node->right) + 1;
		return node;
	}
	
	TreeNode* getMin(TreeNode* node) {
		if (node == NULL) return NULL;
		if (node->left == NULL) return node;
		return getMin(node->left);
	}
	
	TreeNode* getMax(TreeNode* node) {
		if (node == NULL) return NULL;
		if (node->right == NULL) return node;
		return getMax(node->right);
	}
	
	TreeNode* floor(TreeNode* node, int key) {
		if (node == NULL) return NULL;
		if (key < node->key) return floor(node->left, key); // 如果左侧返回 NULL 说明都不符合 那就 NULL 即可
		else if (key == node->key) return node;
		// key > node->key 只有可能在右侧 但是可能右侧也没有符合情况的 那么只能是根结点
		TreeNode* tmp = floor(node->right, key);
		if (tmp == NULL) return node;
		return tmp;
	}
	
	TreeNode* ceil(TreeNode* node, int key) {
		if (node == NULL) return NULL;
		if (key > node->key) return ceil(node->right, key);
		else if (key == node->key) return node;
		TreeNode* tmp = ceil(node->left, key);
		if (tmp == NULL) return node;
		return tmp;
	}
	
	int select(TreeNode* node, int k) {
		if (node == NULL) return -1;
		int t = size(node->left);
		if (t == k) return node->key;
		else if (t < k) return select(node->right, k - t - 1);
		return select(node->left, k);
	}
	
	int rank(TreeNode* node, int key) {
		if (node == NULL) return -1;
		if (key > node->key) return size(node->left) + 1 + rank(node->right, key);
		else if (key < node->key) return rank(node->left, key);
		return size(node->left);
	}
	
	TreeNode* deleteMin(TreeNode* node) {
		if (node->left == NULL) {
			// node 就是要删的结点
			TreeNode* tmp = node->right;
			delete(node);
			return tmp;
		}
		node->left = deleteMin(node->left);
		node->n = size(node->left) + size(node->right) + 1;
		return node;
	}
	
	TreeNode* deleteNode(TreeNode* node, int key) {
		if (node == NULL) return NULL;
		if (key < node->key) node->left = deleteNode(node->left, key);
		else if (key > node->key) node->right = deleteNode(node->right, key);
		else {
			if (node->right == NULL) return node->left;
			if (node->left == NULL) return node->right;
			TreeNode* t = node;
			// t 和 node 互不影响
			node = getMin(t->right);
			node->right = deleteMin(t->right);
			node->left = t->left;
			delete(t);
		}
		node->n = size(node->left) + size(node->right) + 1;
		return node;
	}
	
	// 类似中序遍历
	void keys(TreeNode* node, queue<int>& q, int lo, int hi) {
		if (node == NULL) return;
		if (lo < node->key) keys(node->left, q, lo, hi);
		if (lo <= node->key && node->key <= hi) q.push(node->key);
		if (hi > node->key) keys(node->right, q, lo, hi);
	}
public:
	int size() {
		return size(root);
	}
	
	int getValue(int key) {
		return getValue(root, key);
	}
	
	void put(int key, int value) {
		root = put(root, key, value);
	}
	
	int getMin() {
		TreeNode* t = getMin(root);
		if (t == NULL) return -1;
		return t->key;
	}
	
	int getMax() {
		TreeNode* t = getMax(root);
		if (t == NULL) return -1;
		return t->key;
	}
	
	int floor(int key) {
		TreeNode* tmp = floor(root, key);
		if (tmp == NULL) return -1;
		return tmp->key;
	}
	
	int ceil(int key) {
		TreeNode* tmp = ceil(root, key);
		if (tmp == NULL) return -1;
		return tmp->key;
	}
	
	int select(int k) {
		return select(root, k);
	}
	
	int rank(int key) {
		return rank(root, key);
	}
	
	void deleteMin() {
		root = deleteMin(root);
	}
	
	void deleteNode(int key) {
		root = deleteNode(root, key);
	}
	
	queue<int> keys() {
		return keys(getMin(), getMax());
	}
	
	queue<int> keys(int lo, int hi) {
		queue<int> q;
		keys(root, q, lo, hi);

class AVLTree {
private:
	TreeNode* root;
	
	void updateHeight(TreeNode* &node) {
		if (node == NULL) return;
		node->height = max(getHeight(node->left), getHeight(node->right)) + 1;
	}
	
	// 左旋
	void L(TreeNode* &node) {
		if (node == NULL) return;
		TreeNode* tmp = node->right;
		node->right = tmp->left;
		tmp->left = node;
		updateHeight(node);
		updateHeight(tmp);
		node = tmp;
	}
	
	// 右旋
	void R(TreeNode* &node) {
		if (node == NULL) return;
		TreeNode* tmp = node->left;
		node->left = tmp->right;
		tmp->right = node;
		updateHeight(node);
		updateHeight(tmp);
		node = tmp;
	}
	
	void LL(TreeNode* &node) {
		R(node);
	}
	
	void LR(TreeNode* &node) {
		L(node->left);
		R(node);
	}
	
	void RR(TreeNode* &node) {
		L(node);
	}
	
	void RL(TreeNode* &node) {
		R(node->right);
		L(node);
	}
	
public:
	AVLTree(int data[], int n) {
		root = NULL;
		for (int i = 0; i < n; i++) {
			insert(root, data[i]);
		}
	}
	
	TreeNode* getRoot() {
		return this->root;
	}
	
	int getHeight(TreeNode* node) {
		if (node == NULL) return 0;
		return node->height;

		if (x > node->val) {
			search(node->right, x);

		else if (x < node->val) {

		else {
			// printf("%d is found.", node->val);
			return node;
		}
	}
	
	int getBalanceFactor(TreeNode* node) {
		if (node == NULL) return 0;
		return getHeight(node->left) - getHeight(node->right);
	}

		}

# 堆（优先队列）

const int maxn = 100;
// 大顶堆 下标从 1 开始
int heap[maxn], n = 10;

void swap(int& a, int& b) {
	int temp = a;
	a = b;
	b = temp;
}

// 向下调整 比较 low 和两个子结点 O(logN)
void downAdjust(int low, int high) {
	int i = low, j = i * 2;
	while (j <= high) {
		if (j + 1 <= high && heap[j + 1] > heap[j]) {
			j = j + 1;
		}
		if (heap[j] > heap[i]) {
			swap(heap[i], heap[j]);
			i = j;
			j = i * 2;
		}
		else break;
	}
}

// 向上调整 比较 high 和父节点 O(logN)
void upAdjust(int low, int high) {
	int i = high, j = i / 2;
	while (j >= low) {
		if (heap[i] > heap[j]) {
			swap(heap[i], heap[j]);
			i = j;
			j = i / 2;
		}
		else break;
	}
	/**
	while (i > low && heap[i] > heap[i/2]) {
		swap(heap[i], heap[i/2]);
		i /= 2;
	}
	**/
}

// 建堆 下沉 O(N)
void createHeap() {
	for (int i = n / 2; i >= 1; i--) {
		downAdjust(i, n);
	}
}

// O(logN)
void deleteTop() {
	heap[1] = heap[n--];
	downAdjust(1, n);
}

// 插入 上浮
void insert(int x) {
	heap[++n] = x;
	upAdjust(1, n);
}

// 递增堆排序 下沉
void heapSort() {
	createHeap();
	for (int i = n; i >= 1; i--) {
		swap(heap[i], heap[1]);
		downAdjust(1, i - 1); // 经过之前的排序 [i, n] 已有序（因为每次将堆顶（最大的）移到末尾，所以后面已经是所有最大的元素递增排序）
	}
}

		node = vector<int>(n);
		for (int i = 0;  i < n; i++) {
			node[i] = i;
		}
		size = vector<int>(n, 1);
	}
	
	int find(int x) {
		if (x != node[x]) {
			return node[x] = find(node[x]);
		}
		return x;
	}
	
	void Union(int x, int y) {
		int r1 = find(x), r2 = find(y);
		if (r1 == r2) return;
		if (size[r1] < size[r2]) {
			node[r1] = r2;
			size[r2] += size[r1];
		}
		else {
			node[r2] = r1;
			size[r1] += size[r2];
		}
		count--;
	}
	
	bool isConnected(int x, int y) {
		return find(x) == find(y);
	}
	
	int getCount() {
		return this->count;
	}
};
